Volcano_genes <-EnhancedVolcano(genes,
    lab = genes$`Gene Name`,
    x = 'log2FoldChange',
    y = 'padj',
    xlim = c(-4, 4),
    pCutoff = 0.05,
    ylab= "-Log10 padj value",
    title="Differentially expressed genes \n in Morc3 KO vs WT '\n BioRXiv 2020")
